    Puertas INT,
    Capacidad INT, 
    Cilindrada VARCHAR(20), 
    Patente VARCHAR(10), 
    Disponibilidad VARCHAR(20),
    KM INT, 
    Marca VARCHAR(20), 
    Anio INT

#Crear tabla cliente
CREATE TABLE cliente (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Nombre VARCHAR(40), 
    Edad INT, 
    Deuda INT, 
    FechaDevolucion DATE, 
    ID_Vehiculo INT,
    FOREIGN KEY (ID_Vehiculo) REFERENCES vehiculo(ID) 
        ON DELETE CASCADE 
        ON UPDATE CASCADE
);
